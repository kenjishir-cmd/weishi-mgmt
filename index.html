// --- 核心變數與狀態管理 --- const UNIT_LEFT = ['426-6','426-5','426-4','426-3','426-2','426-1']; const UNIT_RIGHT = ['424-6','424-5','424-4','424-3','424-2','424-1']; const UNITS = []; for (let i = 0; i < 6; i++) { UNITS.push(UNIT_LEFT[i]); UNITS.push(UNIT_RIGHT[i]); } const INIT_BALANCE = 97246; const MONTHS = ['1月','2月','3月','4月','5月','6月','7月','8月','9月','10月','11月','12月']; let state = loadState(); let editingId = null; // 用於追蹤目前正在編輯哪一筆資料 function loadState() { try { const s = localStorage.getItem('weishi-state-v2'); if (s) return JSON.parse(s); } catch(e) {} const payments = {}; UNITS.forEach(u => { payments[u] = {}; MONTHS.forEach(m => payments[u][m] = false); }); return { records: [], payments, nextId: 1 }; } function saveState() { localStorage.setItem('weishi-state-v2', JSON.stringify(state)); } // --- 計算邏輯 --- function totalIncome() { return state.records.filter(r=>r.type==='income').reduce((s,r)=>s+r.amount,0); } function totalExpense() { return state.records.filter(r=>r.type==='expense').reduce((s,r)=>s+r.amount,0); } function balance() { return INIT_BALANCE + totalIncome() - totalExpense(); } function paidUnits() { return UNITS.filter(u => MONTHS.some(m => state.payments[u] && state.payments[u][m])).length; } function fmt(n) { return 'NT$ ' + Math.abs(n).toLocaleString('zh-TW'); } // --- UI 渲染 --- function render() { const b = balance(); const balanceEl = document.getElementById('balance-num'); balanceEl.textContent = b.toLocaleString('zh-TW'); balanceEl.style.color = b >= 0 ? 'var(--gold-light)' : 'var(--red)'; document.getElementById('total-income-disp').textContent = fmt(totalIncome()); document.getElementById('total-expense-disp').textContent = fmt(totalExpense()); document.getElementById('paid-units-disp').textContent = paidUnits() + ' / 12'; document.getElementById('hist-balance').textContent = fmt(b); renderRecords('records-list', 30); renderRecords('history-list', 999); renderUnits(); } function renderRecords(containerId, limit) { const el = document.getElementById(containerId); const records = [...state.records].reverse().slice(0, limit); if (!records.length) { el.innerHTML = `
尚無帳務記錄
點下方按鈕新增收支
`; return; } el.innerHTML = records.map(r => `
${r.desc}
${r.type==='income'?'＋':'－'} ${r.amount.toLocaleString()}
${r.date} ${r.cat}
結餘：NT$ ${r.balanceAfter.toLocaleString()}
`).join(''); } // --- 帳務操作功能 --- // 1. 開啟編輯 function editRecord(id) { const r = state.records.find(x => x.id === id); if (!r) return; editingId = id; const type = r.type; openSheet(type); document.getElementById(type + '-date').value = r.date; document.getElementById(type + '-desc').value = r.desc; document.getElementById(type + '-cat').value = r.cat; document.getElementById(type + '-amount').value = r.amount; // 修改按鈕文字為「更新儲存」 const okBtn = document.querySelector(`#${type}-overlay .btn-ok`); okBtn.textContent = "儲存修改"; } // 2. 刪除紀錄 function deleteRecord(id) { const r = state.records.find(x => x.id === id); if (!r) return; showConfirm('確認刪除', `確定要刪除這筆紀錄嗎？\n${r.desc}\n金額：${fmt(r.amount)}`, () => { state.records = state.records.filter(x => x.id !== id); recalculateBalances(); saveState(); render(); }, true); } // 3. 重新計算歷史結餘（當刪除或修改中間的紀錄時） function recalculateBalances() { let currentBal = INIT_BALANCE; state.records.forEach(r => { currentBal += (r.type === 'income' ? r.amount : -r.amount); r.balanceAfter = currentBal; }); } // 4. 確認新增或更新 function confirmAdd(type) { const date = document.getElementById(type+'-date').value; const desc = document.getElementById(type+'-desc').value.trim(); const cat = document.getElementById(type+'-cat').value; const amount = parseInt(document.getElementById(type+'-amount').value); if (!date || !desc || !amount || amount <= 0) { alert('請填寫完整資料'); return; } const isEditing = editingId !== null; const title = isEditing ? '確認修改' : (type === 'income' ? '確認新增收入' : '確認新增支出'); const msg = `${desc}\nNT$ ${amount.toLocaleString()}\n日期：${date}`; showConfirm(title, msg, () => { if (isEditing) { const idx = state.records.findIndex(x => x.id === editingId); state.records[idx] = { ...state.records[idx], type, date, desc, cat, amount }; editingId = null; } else { state.records.push({ id: state.nextId++, type, date, desc, cat, amount, balanceAfter: 0 }); } recalculateBalances(); saveState(); closeSheet(type); render(); // 恢復按鈕文字 document.querySelector(`#${type}-overlay .btn-ok`).textContent = "確認新增"; }, type === 'expense'); } // --- 通用 UI 控制 --- function switchTab(tab) { document.querySelectorAll('.tab').forEach((t,i) => { t.classList.toggle('active', ['records','units','history'][i] === tab); }); ['records','units','history'].forEach(t => { document.getElementById('tab-'+t).style.display = t===tab ? '' : 'none'; }); document.getElementById('fab-area').style.display = tab === 'units' ? 'none' : 'flex'; } function openSheet(type) { editingId = null; // 重置編輯狀態 const okBtn = document.querySelector(`#${type}-overlay .btn-ok`); if(okBtn) okBtn.textContent = "確認新增"; if (type === 'income' || type === 'expense') { const today = new Date().toISOString().split('T')[0]; document.getElementById(type+'-date').value = today; document.getElementById(type+'-desc').value = ''; document.getElementById(type+'-amount').value = ''; } document.getElementById(type+'-overlay').classList.add('show'); } function closeSheet(type) { document.getElementById(type+'-overlay').classList.remove('show'); editingId = null; } // (其餘原本的 updateClock, renderUnits, showConfirm 等函數保持不變...)
